012/**
013 * The base for all commands. All atomic commands should subclass CommandBase.
014 * CommandBase stores creates and stores each control system. To access a
015 * subsystem elsewhere in your code in your code use CommandBase.exampleSubsystem
016 * @author Author
017 */
018public abstract class CommandBase extends Command {
019
020    public static OI oi;
021    // Create a single static instance of all of your subsystems
022    
023    public static DriveTrain drivetrain = new DriveTrain();
024    public static Shifter shifter = new Shifter ();
025    public static ClawShifter clawShifter = new ClawShifter() ;
026    public static Dumper dumper = new Dumper();
027    
028    
029    
030    public static double autoLeftSpeed[]        = { -0.5, -0.4, -0.5, -0.4, -0.5 };
031    public static double autoRightSpeed[]       = { -0.5, 0.4, -0.5, 0.4, -0.5 };
032    public static double autoDriveTime[]        = { 1.0, 0.55, 0.5, 0.55, 1.0 };
033            
034    public static void init() {
035        // This MUST be here. If the OI creates Commands (which it very likely
036        // will), constructing it during the construction of CommandBase (from
037        // which commands extend), subsystems are not guaranteed to be
038        // yet. Thus, their requires() statements may grab null pointers. Bad
039        // news. Don't move it.
040        oi = new OI();
041
042        // Show what command your subsystem is running on the SmartDashboard
043     
044    }
045
046    public CommandBase(String name) {
047        super(name);
048    }
049
050    public CommandBase() {
051        super();
052    }
053    
054 
055}
