017public class Dumper extends Subsystem {
018    // Put methods for controlling this subsystem
019    // here. Call these from Commands.

021
022    public void initDefaultCommand() {
023        // Set the default command for a subsystem here.
024        //setDefaultCommand(new MySpecialCommand());
025        setDefaultCommand(new MoveDumper());

027    
028    public void move(double speed) {

030    }
